**步骤 3：验证节点结构完整性**
```bash
python -c "
import json
with open('temp/result.json') as f:
    data = json.load(f)
    
# 验证 nodes 存在
if 'nodes' not in data:
    print('[FAIL] No nodes field')
    exit(1)
    
nodes = data['nodes']
print(f'[PASS] nodes array length: {len(nodes)}')

# 验证节点基本字段
required_fields = ['node_class', 'node_name', 'node_pos_x', 'node_pos_y', 'node_guid', 'pins']
for node in nodes[:3]:
    missing = [f for f in required_fields if f not in node]
    if missing:
        print(f'[FAIL] Node {node.get(\"node_name\", \"unknown\")} missing: {missing}')
        exit(1)
    # 验证 GUID 格式（32 字符，无连字符）
    guid = node.get('node_guid', '')
    if guid and (len(guid) != 32 or '-' in guid):
        print(f'[FAIL] GUID format wrong: {guid}')
        exit(1)
        
print('[PASS] First 3 nodes have all required fields + correct GUID format')

# 验证 pins 结构
for node in nodes[:1]:
    pins = node.get('pins', [])
    if not pins:
        print('[WARN] Node has no pins')
        continue
    pin_required = ['pin_id', 'pin_name', 'direction', 'pin_type', 'linked_to']
    for pin in pins[:2]:
        missing = [f for f in pin_required if f not in pin]
        if missing:
            print(f'[FAIL] Pin missing: {missing}')
            exit(1)
    print(f'[PASS] Node {node.get(\"node_name\")} has {len(pins)} pins with required fields')
    
# 验证 FunctionReference
call_func_nodes = [n for n in nodes if 'K2Node_CallFunction' in n.get('node_class', '')]
if call_func_nodes:
    for node in call_func_nodes[:1]:
        if 'function_reference' not in node:
            print(f'[FAIL] K2Node_CallFunction missing function_reference')
            exit(1)
        func_ref = node['function_reference']
        if 'member_name' not in func_ref:
            print(f'[FAIL] function_reference missing member_name')
            exit(1)
    print(f'[PASS] {len(call_func_nodes)} K2Node_CallFunction nodes have function_reference')
    
# 验证 LinkedTo 格式
for node in nodes:
    for pin in node.get('pins', []):
        for conn in pin.get('linked_to', []):
            if 'node_name' not in conn or 'pin_id' not in conn:
                print(f'[FAIL] LinkedTo missing node_name or pin_id')
                exit(1)
                
print('[PASS] All linked_to connections have correct format')
print('[PASS] All validation checks passed')
"
```
